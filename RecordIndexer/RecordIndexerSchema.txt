DROP TABLE if exists Users;
DROP TABLE if exists Projects;
DROP TABLE if exists Fields;
DROP TABLE if exists Images; 
DROP TABLE if exists Valus; 

CREATE TABLE Users 
(
	UserId INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE , 
	UserName TEXT NOT NULL UNIQUE , 
	Email TEXT NOT NULL , 
	FirstName TEXT NOT NULL , 
	LastName TEXT NOT NULL , 
	IndexedRecords INTEGER NOT NULL , 
	PassWord TEXT NOT NULL 
);
CREATE TABLE Projects 
(
	ProjectID INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL UNIQUE ,
	Title TEXT NOT NULL ,	
	RecordSperImage INTEGER NOT NULL ,
	FirstYCoord INTEGER NOT NULL ,
	RecordHeight INTEGER NOT NULL 
);
CREATE TABLE Fields 
(
	FieldID INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE , 
	Title TEXT NOT NULL,
	XCoord INTEGER NOT NULL , 
	Width INTEGER NOT NULL , 
	HelpHTML TEXT NOT NULL , 
	KnownData TEXT , 
	ProjectID INTEGER NOT NULL
);
CREATE TABLE Images 
(
	ImageID INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE , 
	File TEXT NOT NULL , 
	ProjectID INTEGER NOT NULL, 
	UserID INTEGER 
);
CREATE TABLE Valus  ( 	
	ValueID INTEGER PRIMARY KEY  AUTOINCREMENT  NOT NULL  UNIQUE ,  	
	ImageID INTEGER NOT NULL ,  	
	FieldID INTEGER NOT NULL ,  	
	Row INTEGER NOT NULL  ,
	Info TEXT NOT NULL
);
